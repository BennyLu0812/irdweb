CREATE FUNCTION [dbo].[sys_params_split]
    (
      @Str VARCHAR(1000) ,
      @Separator VARCHAR(10)
    )
RETURNS @returntable TABLE
    (
      id INT IDENTITY(1, 1) ,
      value VARCHAR(1000)
    )
AS
    BEGIN
        DECLARE @i INT
        SET @Str = RTRIM(LTRIM(@Str))
        select @Str=PARAM_VALUE from CLP_SYS_PARAMS where PARAM_NAME =@Str
        SET @i = CHARINDEX(@Separator, @Str)
        WHILE @i >= 1
            BEGIN
                INSERT  @returntable
                VALUES  ( LEFT(@Str, @i - 1) )
                SET @Str = SUBSTRING(@Str, @i + 1, LEN(@Str) - @i)
                SET @i = CHARINDEX(@Separator, @Str)
            END
        IF @Str <> '\'
            BEGIN
                INSERT  @returntable
                VALUES  ( @Str )
            END
        RETURN
    END;
